1. Rewrite `deploy/docker-compose.yml` with the following complete content:

```yaml
# Copyright (c) 2025 TRUSTEDGE LABS LLC
# MPL-2.0: https://mozilla.org/MPL/2.0/
# Project: trustedge — Privacy and trust at the edge.

# TrustEdge Platform — Full Stack
#
# Starts postgres, platform-server, and dashboard with a single command.
# No .env file needed — all defaults are inline for demo use.
#
# Usage:
#   docker compose -f deploy/docker-compose.yml up --build

services:
  postgres:
    image: postgres:16-alpine
    environment:
      POSTGRES_USER: trustedge
      POSTGRES_PASSWORD: trustedge_dev
      POSTGRES_DB: trustedge
    volumes:
      - postgres_data:/var/lib/postgresql/data
    ports:
      - "5432:5432"
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U trustedge"]
      interval: 5s
      timeout: 5s
      retries: 5

  platform-server:
    build:
      context: ..
      dockerfile: deploy/Dockerfile
    environment:
      DATABASE_URL: postgres://trustedge:trustedge_dev@postgres:5432/trustedge
      PORT: "3001"
    ports:
      - "3001:3001"
    depends_on:
      postgres:
        condition: service_healthy
    healthcheck:
      test: ["CMD-SHELL", "wget -qO- http://localhost:3001/healthz || exit 1"]
      interval: 10s
      timeout: 5s
      retries: 5
      start_period: 5s
    command: ["sh", "-c", "trustedge-platform-server migrate && trustedge-platform-server serve"]
    restart: unless-stopped

  dashboard:
    build:
      context: ..
      dockerfile: deploy/Dockerfile.dashboard
      args:
        VITE_API_BASE: http://localhost:3001
    ports:
      - "8080:80"
    depends_on:
      platform-server:
        condition: service_healthy
    healthcheck:
      test: ["CMD-SHELL", "wget -qO- http://localhost/healthz || exit 1"]
      interval: 10s
      timeout: 3s
      retries: 3
    restart: unless-stopped

volumes:
  postgres_data:
```
