**action.yml** — Composite action:
```yaml
name: 'TrustEdge SBOM Attestation'
description: 'Attest a binary artifact with its CycloneDX SBOM using TrustEdge'
inputs:
  binary:
    description: 'Path to the binary artifact to attest'
    required: true
  sbom:
    description: 'Path to CycloneDX JSON SBOM file'
    required: true
  key:
    description: 'Path to Ed25519 device key (generates ephemeral if not provided)'
    required: false
    default: ''
  trst-version:
    description: 'TrustEdge release version to download (e.g., v4.0.0)'
    required: false
    default: 'latest'
outputs:
  attestation-path:
    description: 'Path to generated .te-attestation.json file'
    value: ${{ steps.attest.outputs.attestation-path }}
runs:
  using: composite
  steps:
    - name: Download trst binary
      shell: bash
      run: |
        VERSION="${{ inputs.trst-version }}"
        if [ "$VERSION" = "latest" ]; then
          URL="https://github.com/TrustEdge-Labs/trustedge/releases/latest/download/trst"
        else
          URL="https://github.com/TrustEdge-Labs/trustedge/releases/download/${VERSION}/trst"
        fi
        curl -fsSL "$URL" -o "${{ runner.temp }}/trst"
        chmod +x "${{ runner.temp }}/trst"
        echo "Downloaded trst from $URL"

    - name: Generate ephemeral keypair (if no key provided)
      shell: bash
      if: inputs.key == ''
      run: |
        "${{ runner.temp }}/trst" keygen \
          --out-key "${{ runner.temp }}/ephemeral.key" \
          --out-pub "${{ runner.temp }}/ephemeral.pub" \
          --unencrypted
        echo "TRST_KEY=${{ runner.temp }}/ephemeral.key" >> "$GITHUB_ENV"
        echo "TRST_PUB=${{ runner.temp }}/ephemeral.pub" >> "$GITHUB_ENV"

    - name: Set provided key paths
      shell: bash
      if: inputs.key != ''
      run: |
        echo "TRST_KEY=${{ inputs.key }}" >> "$GITHUB_ENV"
        # Derive pub path (same dir, .pub extension)
        PUB_PATH="${{ inputs.key }}"
        PUB_PATH="${PUB_PATH%.key}.pub"
        echo "TRST_PUB=${PUB_PATH}" >> "$GITHUB_ENV"

    - name: Create SBOM attestation
      id: attest
      shell: bash
      run: |
        BINARY_NAME=$(basename "${{ inputs.binary }}")
        OUT_PATH="${{ runner.temp }}/${BINARY_NAME}.te-attestation.json"
        "${{ runner.temp }}/trst" attest-sbom \
          --binary "${{ inputs.binary }}" \
          --sbom "${{ inputs.sbom }}" \
          --device-key "$TRST_KEY" \
          --device-pub "$TRST_PUB" \
          --out "$OUT_PATH" \
          --unencrypted
        echo "attestation-path=$OUT_PATH" >> "$GITHUB_OUTPUT"
        echo "Created attestation: $OUT_PATH"
```

1. **Job name:** `self-attestation`
2. **Condition:** `if: startsWith(github.ref, 'refs/tags/v')`
3. **Needs:** `build-and-test` (only attest after tests pass)
4. **Permissions:** `contents: write` (needed for uploading release assets)

Steps:
1. Checkout repo
2. Install Rust toolchain (using dtolnay/rust-toolchain, SHA-pinned, matching existing pattern)
3. Use rust-cache (matching existing pattern)
4. Build trst binary: `cargo build -p trustedge-trst-cli --release --locked`
5. Install syft:
   ```bash
   curl -sSfL https://raw.githubusercontent.com/anchore/syft/main/install.sh | sh -s -- -b /usr/local/bin
   ```
6. Generate ephemeral Ed25519 keypair (per D-05):
   ```bash
   ./target/release/trst keygen --out-key ephemeral.key --out-pub ephemeral.pub --unencrypted
   ```
7. Generate CycloneDX SBOM with syft (per D-07):
   ```bash
   syft ./target/release/trst -o cyclonedx-json > trst-sbom.cdx.json
   ```
8. Create attestation (per D-06):
   ```bash
   ./target/release/trst attest-sbom \
     --binary ./target/release/trst \
     --sbom trst-sbom.cdx.json \
     --device-key ephemeral.key \
     --device-pub ephemeral.pub \
     --out trst.te-attestation.json \
     --unencrypted
   ```
9. Upload attestation as release asset using `gh release upload`:
   ```bash
   gh release upload "${{ github.ref_name }}" trst.te-attestation.json --clobber
   ```
   Set `GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}` in env.
