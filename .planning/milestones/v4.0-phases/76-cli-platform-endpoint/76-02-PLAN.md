```rust
/// POST /v1/verify-attestation — verify a point attestation document.
///
/// Accepts the attestation JSON document directly as the request body (per D-11).
/// Extracts the embedded public key, verifies the signature, and returns a JWS receipt
/// on success. No database interaction — works identically with or without postgres.
pub async fn verify_attestation_handler(
    State(state): State<AppState>,
    body: String,
) -> Result<Json<VerifyAttestationResponse>, (StatusCode, Json<ValidationError>)> {
```

Handler logic:
- Parse body as PointAttestation via `PointAttestation::from_json(&body)`. On failure, return 400 with ValidationError("invalid_attestation", "Failed to parse attestation document").
- Check `attestation.format == FORMAT_V1`. If not, return 400 with ValidationError("invalid_format", "Expected format te-point-attestation-v1"). (per D-15)
- Check `attestation.signature.is_some()`. If None, return 400 with ValidationError("missing_signature", "Attestation document has no signature").
- Extract public_key from attestation: `let device_pub = &attestation.public_key;` (per D-12).
- Call `attestation.verify_signature(device_pub)`. On crypto error, return 200 with status "failed" and generic error detail "Cryptographic verification failed" (per D-17 — no internal details leaked).
- If signature valid:
  - Generate verification_id: `format!("va_{}", uuid::Uuid::new_v4().simple())`
  - Build ReceiptClaims with: verification_id, device_id = attestation.public_key (truncated or full), manifest_digest = BLAKE3 hash of canonical bytes (`blake3::hash(&attestation.canonical_bytes().unwrap_or_default()).to_hex()`), chain_tip = "none" (not applicable for point attestations), timestamp = current UTC RFC3339, kid from key_manager, result = a VerifyReport with signature "pass".
  - NOTE: ReceiptClaims.result is VerifyReport which has fields specific to .trst archives. For point attestations, populate it minimally: signature="pass", continuity="n/a", segments=0, profile="point-attestation", device_id=attestation.public_key. The important receipt data is in the JWS claims themselves.
  - Call `sign_receipt_jws(&receipt_claims, &keys, state.receipt_ttl_secs).await`. On error, return 500 with ValidationError("receipt_error", "Failed to sign receipt").
  - Return 200 with `VerifyAttestationResponse { status: "verified", receipt: Some(jws_token), details }` where details is a JSON object containing: subject_hash, evidence_hash, public_key, timestamp, format. (per D-13)
- If signature invalid:
  - Return 200 with `VerifyAttestationResponse { status: "failed", receipt: None, details }` where details contains: reason = "Signature verification failed", public_key, timestamp. (per D-13, D-17)

```rust
fn build_test_attestation(signing_key: &ed25519_dalek::SigningKey) -> String {
    // Create temp files for hashing
    let dir = tempfile::tempdir().unwrap();
    let binary_path = dir.path().join("test.bin");
    let sbom_path = dir.path().join("sbom.json");
    std::fs::write(&binary_path, b"test binary content").unwrap();
    std::fs::write(&sbom_path, r#"{"bomFormat":"CycloneDX","components":[]}"#).unwrap();

    let attestation = trustedge_core::PointAttestation::create(
        &binary_path, "binary", &sbom_path, "sbom", &keypair,
    ).unwrap();
    attestation.to_json().unwrap()
}
```

```rust
async fn test_verify_attestation_valid() {
    let app = create_test_app().await;
    let signing_key = SigningKey::generate(&mut OsRng);

    let request = Request::builder()
```
