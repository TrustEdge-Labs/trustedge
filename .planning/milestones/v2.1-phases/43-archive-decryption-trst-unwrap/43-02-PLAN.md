       ```rust
       struct UnwrapCmd {
           #[arg(value_name = "ARCHIVE", help = "Path to .trst archive directory")]
           archive: PathBuf,
           #[arg(
               long = "device-key",
               value_name = "PATH",

           #[arg(long = "out", value_name = "PATH", help = "Output file path for recovered data")]
           output: PathBuf,
       }
       ```

    2. Add `Unwrap(UnwrapCmd)` variant to the `Commands` enum.

    3. Add match arm in `run()`: `Commands::Unwrap(args) => handle_unwrap(args),`

    4. Implement `handle_unwrap(args: UnwrapCmd) -> Result<()>`:

       a. Load device keypair from file:
          ```rust
          let key_bytes = fs::read(&args.device_key)
              .with_context(|| format!("failed to read device key '{}'", args.device_key.display()))?;
          let contents = String::from_utf8_lossy(&key_bytes).trim().to_string();
          let device_keypair = DeviceKeypair::import_secret(&contents)?;
          ```

       b. Read archive:
          ```rust
          let (manifest, chunks) = read_archive(&args.archive)
              .with_context(|| format!("Failed to read archive: {}", args.archive.display()))?;
          ```

       c. Verify signature BEFORE any decryption:
          ```rust
          let signature = manifest.signature.as_ref()
              .ok_or_else(|| anyhow::anyhow!("Archive has no signature"))?;
          let canonical_bytes = manifest.to_canonical_bytes()?;
          let sig_valid = verify_manifest(&device_keypair.public, &canonical_bytes, signature)?;
          if !sig_valid {
              eprintln!("Signature: FAIL");
              process::exit(10);
          }
          eprintln!("Signature: PASS");
          ```

       d. Validate continuity chain:
          ```rust
          validate_archive(&args.archive)
              .map_err(|e| {
                  eprintln!("Continuity: FAIL ({})", e);
                  process::exit(11);
              }).ok();
          eprintln!("Continuity: PASS");
          ```
          Note: validate_archive re-reads the archive internally. This is acceptable for correctness (it uses its own read_archive + chain validation). Alternatively, to avoid double-read, replicate the chain validation inline. For simplicity and correctness, use validate_archive() which is the canonical verification path.

       e. Derive chunk key:
          ```rust
          let encryption_key = derive_chunk_key(device_keypair.secret_bytes());
          ```

       f. Extract started_at from metadata for AAD reconstruction. Add a helper or match:
          ```rust
          let started_at = match &manifest.metadata {
              ProfileMetadata::CamVideo(m) => &m.started_at,
              ProfileMetadata::Sensor(m) => &m.started_at,
              ProfileMetadata::Audio(m) => &m.started_at,
              ProfileMetadata::Log(m) => &m.started_at,
              ProfileMetadata::Generic(m) => &m.started_at,
          };
          ```

       g. Decrypt chunks in order and reassemble:
          ```rust
          let aad = generate_aad(&manifest.trst_version, &manifest.profile, &manifest.device.id, started_at);
          let mut output_data = Vec::new();

          for (index, chunk_bytes) in &chunks {
              if chunk_bytes.len() < 24 {
                  anyhow::bail!("Chunk {:05} too short to contain nonce ({} bytes)", index, chunk_bytes.len());
              }
              let nonce: [u8; 24] = chunk_bytes[..24].try_into().unwrap();
              let ciphertext = &chunk_bytes[24..];
              let plaintext = decrypt_segment(&encryption_key, &nonce, ciphertext, &aad)
                  .with_context(|| format!("Failed to decrypt chunk {:05}", index))?;
              output_data.extend_from_slice(&plaintext);
          }
          ```

       h. Write output file:
          ```rust
          fs::write(&args.output, &output_data)
              .with_context(|| format!("Failed to write output: {}", args.output.display()))?;
          ```

       i. Print summary to stderr:
          ```rust
          eprintln!("Chunks: {}", chunks.len());
          eprintln!("Bytes: {}", output_data.len());
          eprintln!("Output: {}", args.output.display());
          ```

    5. Add `decrypt_segment` and `validate_archive` to the `use trustedge_core::{...}` import if not already present. Also add `ProfileMetadata` variants needed for the started_at match (they should already be imported from Plan 01 context — CamVideoMetadata etc are not needed, just ProfileMetadata which is already imported).

    6. CRITICAL: On any verification failure (signature or continuity), the function MUST exit without writing any output file. The exit codes are: 10 for signature failure, 11 for continuity failure. Use `process::exit()` after printing the error, matching the existing verify command pattern.

    7. On decryption failure (wrong key will cause AEAD auth tag mismatch), print an error message like "Decryption failed — wrong device key or corrupted archive" and exit with code 1. Do NOT write any partial output.
