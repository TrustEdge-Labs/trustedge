```rust
// TRUSTEDGE-KEY-V1\n
// {"salt":"<b64>","nonce":"<b64>","iterations":600000}\n

const ENCRYPTED_KEY_HEADER: &str = "TRUSTEDGE-KEY-V1";

impl DeviceKeypair {
    pub fn generate() -> Result<Self, CryptoError>;
    pub fn export_secret_encrypted(&self, passphrase: &str) -> Result<Vec<u8>, CryptoError>;
    pub fn import_secret_encrypted(data: &[u8], passphrase: &str) -> Result<Self, CryptoError>;
}

pub fn is_encrypted_key_file(data: &[u8]) -> bool;
```

From crates/core/src/error.rs:
```rust
pub enum CryptoError {
    InvalidKeyFormat(String),   // Returned for truncated/corrupted format issues
    DecryptionFailed(String),   // Returned for wrong passphrase or corrupted ciphertext
    EncryptionFailed(String),
    // ...
}
```

Error paths in import_secret_encrypted():
- No newline found -> InvalidKeyFormat("Missing header line")
- Non-UTF8 header -> InvalidKeyFormat("Invalid UTF-8 header")
- Wrong header string -> InvalidKeyFormat("Expected header '...', got '...'")
- No second newline -> InvalidKeyFormat("Missing metadata line")
- Non-UTF8 metadata -> InvalidKeyFormat("Invalid UTF-8 metadata")
- Bad JSON -> InvalidKeyFormat("Invalid JSON metadata: ...")
- Missing salt/nonce/iterations -> InvalidKeyFormat("Missing salt/nonce/iterations")
- Bad base64 -> InvalidKeyFormat("Invalid salt/nonce base64: ...")
- Wrong nonce length -> InvalidKeyFormat("Nonce must be 12 bytes")
- AES-GCM decrypt fails -> DecryptionFailed("Wrong passphrase or corrupted key file")
- Decrypted key != 32 bytes -> InvalidKeyFormat("Decrypted key must be 32 bytes")
